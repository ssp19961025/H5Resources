			var s2='123abc'

			var n2=parseInt(s2);
			
			console.log(n2+4)  //127
			
			//不可以转化
			
			var s3='a123abc';
			var n3=parseInt(s3);   //NAN

10. 回调函数

		//把函数作为实参传给调用的函数，在调用的函数中再调用之前传给它的那个函数
		function add(a,b,callback) {
		    var c=a+b;
		    callback(c);   //show(3)
		}
		
		
		function show(res) {
		    console.log(res);
		}
		
		add(1,2,show);
		
		add(4,5,function (res) {
		    console.log(res);
		});
		
		add(4,5,res=>{
		    console.log(res);
		});
